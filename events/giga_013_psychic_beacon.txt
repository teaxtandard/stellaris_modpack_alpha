# Activate beacon
country_event = {
	id = giga_psychic_beacon.001
	title = "giga_psychic_beacon.001.name"
	desc = "giga_psychic_beacon.001.desc"
	is_triggered_only = yes
	show_sound = event_mystic_reveal

	diplomatic = yes

	picture_event_data = {
		portrait = shroud1
		room = shroud_room
	}

	option = { # Illuminate (Change planet class from shrouded to lifeless)
		name = "giga_psychic_beacon.001.a"
		custom_tooltip = "giga_psychic_beacon.001.a.tooltip"
		allow = {
			resource_stockpile_compare = { resource = sr_psionic_sublimate value >= 20 }
			custom_tooltip = {
				fail_text = "can_already_illuminate"
				NOT = { has_country_flag = giga_psychic_beacon_illumiate }
			}
			custom_tooltip = {
				fail_text = "must_have_shrouded_world"
				any_planet_within_border = {
					OR = {
						is_planet_class = pc_shrouded
						has_modifier = beacon_shrouded_habitable_structure
					}
				}
			}
		}
		add_resource = { sr_psionic_sublimate = -20 }
		hidden_effect = { set_country_flag = giga_psychic_beacon_illumiate }
	}

	option = { # Traverse (+1000% jumpdrive range)
		name = "giga_psychic_beacon.001.b"
		custom_tooltip = "giga_psychic_beacon.001.b.tooltip"
		allow = {
			resource_stockpile_compare = { resource = sr_psionic_sublimate value >= 30 }
			NOT = { has_modifier = "giga_psychic_traverse" }
		}
		add_resource = { sr_psionic_sublimate = -50 }
		add_modifier = { modifier = "giga_psychic_traverse" days = 900 }
		hidden_effect = { country_event = { id = giga_psychic_beacon.009 } }
	}

	option = { # Dominate (Convert pops on hostile planets to psionic)
		name = "giga_psychic_beacon.001.c"
		custom_tooltip = "giga_psychic_beacon.001.c.tooltip"
		allow = {
			resource_stockpile_compare = { resource = sr_psionic_sublimate value >= 200 }
			custom_tooltip = {
				fail_text = "must_be_at_war_and_own_systems"
				any_country = {
					is_at_war_with = root
					OR = {
						ehof_default_country = yes
						is_country_type = katzenartig_imperium
					}
					any_planet_within_border = {
						OR = {
							is_owned_by = ROOT
							is_controlled_by = ROOT
						}
					}
				}
			}
		}
		add_resource = { sr_psionic_sublimate = -200 }
		hidden_effect = { country_event = { id = giga_psychic_beacon.003 } }
	}

	option = { # Imprison (Any hostile colony > shrouded world)
		name = "giga_psychic_beacon.001.d"
		custom_tooltip = "giga_psychic_beacon.001.d.tooltip"
		allow = {
			resource_stockpile_compare = { resource = sr_psionic_sublimate value >= 1000 }
			custom_tooltip = {
				fail_text = "must_be_at_war"
				OR = {
					any_country = {
						is_at_war_with = root
						OR = {
							ehof_default_country = yes
							is_country_type = katzenartig_imperium
						}
					}
					any_country = {
						has_country_flag = paluush_primitives
					}
				}
			}
		}
		add_resource = { sr_psionic_sublimate = -1000 }
		hidden_effect = { country_event = { id = giga_psychic_beacon.005 } }
	}

	option = { # Conjure (Spawn psionic armies on enemy world)
		name = "giga_psychic_beacon.001.e"
		custom_tooltip = "giga_psychic_beacon.001.e.tooltip"
		allow = {
			resource_stockpile_compare = { resource = sr_psionic_sublimate value >= 60 }
			custom_tooltip = {
				fail_text = "must_be_at_war"
				OR = {
					any_country = {
						is_at_war_with = root
						OR = {
							ehof_default_country = yes
							is_country_type = katzenartig_imperium
						}
					}
					any_country = {
						has_country_flag = paluush_primitives
					}
				}
			}
		}
		add_resource = { sr_psionic_sublimate = -60 }
		hidden_effect = { country_event = { id = giga_psychic_beacon.010 } }
	}

	option = { # Imprint (Buff admiral)
		name = "giga_psychic_beacon.001.f"
		custom_tooltip = "giga_psychic_beacon.001.f.tooltip"
		allow = {
			resource_stockpile_compare = { resource = sr_psionic_sublimate value >= 100 }
			custom_tooltip = {
				fail_text = "must_have_valid_beacon_admiral"
				any_owned_leader = {
					leader_class = admiral
					NAND = {
						has_trait = leader_trait_admiral_astute
						has_trait = leader_trait_admiral_farsighted
						has_trait = leader_trait_admiral_fortifier
					}
				}
			}
		}
		add_resource = { sr_psionic_sublimate = -100 }
		hidden_effect = { country_event = { id = giga_psychic_beacon.015 } }
	}

	option = { # Exit
		default_hide_option = yes
		name = "giga_psychic_beacon.001.g"
	}

	after = {
		random_owned_megastructure = {
			limit = { is_megastructure_type = psychic_beacon_4 }
			solar_system = { spawn_megastructure = { type = "psychic_beacon_3" location = prev.planet owner = root } }
			remove_megastructure = this
		}
	}
}
